\chapter{Matrices}

\section{Row matrix}
\begin{enumerate}
    \item Row matrices whose delimiters are of the default size---and not too big.
    \begin{center}
        \begin{tabular}{cC}
            Command & \text{Output}\\
            \verb|\row{1,2}| & \row{1,2}\\
            \verb|\row[delim=p]{1,2}| & \row[delim=p]{1,2}\\
            \verb|\row[delim=b]{1,2}| & \row[delim=b]{1,2}\\
            \verb|\row[delim=B]{1,2}| & \row[delim=B]{1,2}\\
            \verb|\row[delim=v]{1,2}| & \row[delim=v]{1,2}\\
            \verb|\row[delim=V]{1,2}| & \row[delim=V]{1,2}\\
        \end{tabular}
    \end{center}
    \item Row matrices with default sized delimiters, which are too small.
    \begin{center}
        \renewcommand{\arraystretch}{1.8}
        \begin{tabular}{cC}
            Command & \text{Output}\\
            \verb|\row{\dfrac{1}{2}, \dfrac{1}{3}}| & \row{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=p]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=p]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=b]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=b]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=B]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=B]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=v]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=v]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=V]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=V]{\dfrac{1}{2}, \dfrac{1}{3}}\\
        \end{tabular}
    \end{center}
    \item Row matrices with scaled delimiters.
    \begin{center}
        \renewcommand{\arraystretch}{2.0}
        \begin{tabular}{cCC}
            Command & \text{Output} & \text{Correct output}\\
            \verb|\row*{\dfrac{1}{2}, \dfrac{1}{3}}| & \row*{\dfrac{1}{2}, \dfrac{1}{3}} & \begin{pmatrix} \dfrac{1}{2} & \dfrac{1}{3} \end{pmatrix}\\
            \verb|\row*[delim=p]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row*[delim=p]{\dfrac{1}{2}, \dfrac{1}{3}} & \begin{pmatrix} \dfrac{1}{2} & \dfrac{1}{3} \end{pmatrix}\\
            \verb|\row*[delim=b]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row*[delim=b]{\dfrac{1}{2}, \dfrac{1}{3}} & \begin{bmatrix} \dfrac{1}{2} & \dfrac{1}{3} \end{bmatrix}\\
            \verb|\row*[delim=B]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row*[delim=B]{\dfrac{1}{2}, \dfrac{1}{3}} & \begin{Bmatrix} \dfrac{1}{2} & \dfrac{1}{3} \end{Bmatrix}\\
            \verb|\row*[delim=v]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row*[delim=v]{\dfrac{1}{2}, \dfrac{1}{3}} & \begin{vmatrix} \dfrac{1}{2} & \dfrac{1}{3} \end{vmatrix}\\
            \verb|\row*[delim=V]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row*[delim=V]{\dfrac{1}{2}, \dfrac{1}{3}} & \begin{Vmatrix} \dfrac{1}{2} & \dfrac{1}{3} \end{Vmatrix}\\
        \end{tabular}
    \end{center}
    \item Row matrices with manually scaled delimiters.
    \begin{center}
        \renewcommand{\arraystretch}{1.8}
        \begin{tabular}{cC}
            Command & \text{Output}\\
            \verb|\row[delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=p,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=p,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=b,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=b,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=B,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=B,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=v,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=v,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}\\
            \verb|\row[delim=V,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}| & \row[delim=V,delim-size=\Big]{\dfrac{1}{2}, \dfrac{1}{3}}\\
        \end{tabular}
    \end{center}
\end{enumerate}
