\ExplSyntaxOn
% The command \CreateParent{ name }{ clist of Pagestyles }{ clist of corresponding values for the keytheorems Parent key} that defines a macro \<name>Parent, such that 'Parent = \<name>Parent' automatically adapts to the current \Pagestyle.
\int_new:N \l__preamble_CreateParent_number_in_Pagestyles_int
\clist_new:N \l__preamble_CreateParent_all_Pagestyles_clist
\clist_new:N \l__preamble_CreateParent_parent_values_clist
\NewDocumentCommand{\CreateParent}{ m m m }{
    % Initalising the macro that is the value for the keytheorems key 'Parent = '
    \tl_new:c {#1Parent}
    % Defining #2 and #3 semantically as a clist
    \clist_set:Nn \l__preamble_CreateParent_all_Pagestyles_clist {#2}
    \clist_set:Nn \l__preamble_CreateParent_parent_values_clist {#3}
    % Setting the value in "parent = <value>" for the current Pagestyle
    \int_zero:N \l__preamble_CreateParent_number_in_Pagestyles_int
    \clist_map_inline:nn {#2} 
    {
        \int_incr:N \l__preamble_CreateParent_number_in_Pagestyles_int
        \tl_if_eq:NnT \g__preamble_pagestyle_tl {##1} 
        {
            \tl_gset:ce {#1Parent} 
            {
                \clist_item:Nn \l__preamble_CreateParent_parent_values_clist {\l__preamble_CreateParent_number_in_Pagestyles_int}
            }
        }
    }
}
